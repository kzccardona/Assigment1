CREATE TABLE FACTURAS(
ID INT,
IDUSUARIO INT,
IDCONDUCTOR INT,
IDVEHICULO INT,
IDMETODODEPAGO INT,
IDCUPON INT,
IDMONEDA INT,
IDCIUDAD INT,
VALOR DECIMAL(20,2),
DESCUENTOS DECIMAL(20,2),
VALORNETO DECIMAL(20,2),
HORAINICIO TIMESTAMP,
HORAFIN TIMESTAMP
);

CREATE TABLE DETALLEFACTURAS(
ID INT,
IDFACTURA INT,
IDRECORRDO INT,
ORIGEN VARCHAR2(255),
DESTINO VARCHAR2(255),
MILLAS DECIMAL(20,2),
TIEMPO VARCHAR2(255)
);

CREATE TABLE RECORRIDOS(
ID INT,
IDDETALLEFACTURA INT,
IDVEHICULOCONDUCTOR INT,
LATITUD DECIMAL(40,2),
LONGITUD DECIMAL(40,2),
FECHA TIMESTAMP
);

CREATE TABLE USUARIOS(
ID INT,
IDPAIS INT,
NOMBRES VARCHAR2(255),
APELLIDOS VARCHAR2(255),
CELULAR VARCHAR2(255),
EMAIL VARCHAR2(255),
EMAIL2 VARCHAR2(255),
FOTO VARCHAR2(255)
);

CREATE TABLE EMPRESAS(
ID INT,
IDPAIS INT,
IDENVIAREMAIL INT,
NOMBRE VARCHAR2(255),
CELULAR VARCHAR2(255),
EMAIL VARCHAR2(255),
EMAIL2 VARCHAR2(255)
);

CREATE TABLE PAISES (
ID INT,
IDMONEDA INT,
NOMBRE VARCHAR2(255),
IDENTIFICADOR VARCHAR2(255)
);

CREATE TABLE CIUDADES (
ID INT,
IDPAIS INT,
NOMBRE VARCHAR2(255)
);

CREATE TABLE MONEDAS (
ID INT,
NOMBRE VARCHAR2(255),
ABREVIATURA VARCHAR2(255)
);

CREATE TABLE PAGOS(
ID INT,
IDPROPIETARIO INT,
IDPROPIETARIOBANCO INT,
VALOR DECIMAL(20,2),
DESCUENTOS DECIMAL(20,2),
INCENTIVOS DECIMAL(20,2),
VALORNETO DECIMAL(20,2),
FECHA DATE
);

CREATE TABLE DETALLEPAGOS(
ID INT,
IDPAGO INT,
IDRECORRIDO INT,
VALOR DECIMAL(20,2),
DESCUENTOS DECIMAL(20,2),
INCENTIVOS DECIMAL(20,2),
VALORNETO DECIMAL(20,2)
);
